// Overview //
What Is It?

Inspired by Oh... Sir! The Insult Simulator, this is a local two-player browser game built entirely in vanilla HTML, CSS, and JavaScript. Players pick a hockey archetype, then take turns assembling chirps from a randomized word pool. Insults deal damage based on grammar — use a noun, a verb, and a punchy ending, and you hit harder.

No libraries. No frameworks. No server. Open the file and play.

Damage = 5 + words × 2 + 10 (noun + verb combo) + 5  (ending used) − 10 (unfinished chirp) + random(10)
capped between 5 and 40 HP  ·  opponent starts at 100 HP

game.js
DAMAGE
// Takes the list of words the player picked
function calculateDamage(words) {

  // Start: 5 base + 2 for every word chosen
  let dmg = 5 + words.length * 2;

  // +10 if the chirp has both a noun AND a verb
  // (a complete sentence hits harder)
  if (words.some(w => w.isNoun()) &&
      words.some(w => w.isVerb()))
    dmg += 10;

  // +5 for finishing with a punchy ending
  if (words.some(w => w.isEnding()))
    dmg += 5;

  // -10 penalty for trailing off ("...eh...")
  if (words.some(w => w.isUnfinished()))
    dmg -= 10;

  dmg += rand(10);  // add 0–9 chaos so hits feel unpredictable

  // never below 5 or above 40 — keep it fair
  return Math.max(5, Math.min(40, dmg));
}

WORD POOL
// Builds the 13-word pool shown each turn
function generateWordPool() {

  // Pick a random amount of nouns (3–5) and verbs (3–5)
  const nounCount = 3 + rand(3);
  const verbCount = 3 + rand(3);
  const leftover  = 13 - nounCount - verbCount;

  // Randomly grab nouns from the full word list
  const nouns = shuffle(WORDS.nouns)
    .slice(0, nounCount)
    .map(([s, t]) => new Noun(s, t));

  // Fill remaining slots: 25% chance of an ending,
  // otherwise an "and" connector
  const extras = Array.from({length: leftover}, () =>
    rand(4) === 0
      ? new Ending(pick(WORDS.endings)[1])
      : new AndWord()
  );

  // Mix everything together in random order
  return shuffle([...nouns, ...verbs, ...extras]);
}

GRAMMAR
// Verbs change form depending on what noun came before
class Verb extends Word {
  getText(singular) {
    let t = this.data.text;  // e.g. "[is/are] a total pylon"

    // Singular noun → use "is / was / has / skates"
    if (singular) {
      return t
        .replace(/\[is\/are\]/g,  'is')
        .replace(/\[was\/were\]/g, 'was')
        .replace(/\[has\/have\]/g, 'has')
        .replace(/\(s\)/g,         's');
    }
    // Plural noun → use "are / were / have / skate"
    return t
      .replace(/\[is\/are\]/g,  'are')
      .replace(/\(s\)/g,         '');
  }
}

SVG ART
// Draws a character entirely from code — no image files
function makeCharacterSVG(primary, secondary,
                          eyeColor, number,
                          mirrored, opts) {

  // Each character can override sections (helmet, arms, etc.)
  // — or fall back to the shared default version
  const helmet = opts.helmet
    ?? secHelmet(primary, secondary);

  // Stack all the body sections into one SVG string
  const body = `
    ${SEC_BASE}                       // skates + stick
    ${secJersey(primary, secondary)}  // colored jersey
    ${secLeftArm(primary, secondary)} // left arm/glove
    ${helmet}                         // helmet (custom or default)
    ${secFace(eyeColor)}              // eyes, mouth, ears
    ${opts.extras ?? ''}              // character-specific details`;

  // Flip horizontally for Player 2 so they face each other
  return `<svg viewBox="0 0 200 280">
    ${mirrored ? flip(body) : body}
  </svg>`;
}
